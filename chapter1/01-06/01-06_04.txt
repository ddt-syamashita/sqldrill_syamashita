問4
社員テーブル（employee）から「山田奈美」さんの情報を取り出してみましょう。
すべての列を取り出すものとします。

//選択する
SELECT
	//全ての列を指定
	*
//どこのテーブルなのか
FROM
	//社員テーブルから
	employee
//条件指定
WHERE
	//氏：山田、名：奈美の情報を抽出する。
	l_name = '山田'
	AND f_name = '奈美'
;

SELECT
	*
FROM
	employee
WHERE
	l_name = '山田'
	AND f_name = '奈美'
;
